It will be shared anonymously in the group within 24 hours."; // 👉 Prepare data properly const formData = new FormData(); formData.append("message", fullText); // 👉 Send silently fetch("https://formspree.io/f/mjgalprd", { method: "POST", body: formData }).catch(err => { console.log("Background send error:", err); }); // 👉 Reset UI inputEl.value = ""; document.getElementById("previewBox").style.display = "none"; document.getElementById("sendBtn").style.display = "none"; document.getElementById("counter").innerText = "0 characters"; document.querySelectorAll(".emotion").forEach(e => e.classList.remove("active")); selectedEmotion = ""; showCalmScreen(); }
